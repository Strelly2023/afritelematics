# -------------------------------------------------------------------
# GLOBAL EPISTEMIC INVARIANTS
# -------------------------------------------------------------------

global_invariants:

  - id: NO_SCALAR_CONFIDENCE
    description: >
      Scalar or collapsed confidence values are forbidden.
      Confidence MUST be decomposed into explicit dimensions.

  - id: EVIDENCE_BOUND_ONLY
    description: >
      Confidence MUST derive exclusively from attestable,
      replayable inputs.
      Model self-assessed confidence is forbidden.

  - id: REPLAY_REPRODUCIBILITY_REQUIRED
    description: >
      Every confidence dimension MUST be recomputable
      during replay under identical conditions.

  - id: NORMALIZATION_REQUIRED
    description: >
      All confidence dimensions MUST be normalized
      onto a common numeric scale.

  - id: UNCERTAINTY_IS_LEGITIMATE
    description: >
      Low confidence values are acceptable.
      Absence of confidence inflation is not a failure.

# -------------------------------------------------------------------
# NORMALIZATION RULES
# -------------------------------------------------------------------

normalization:

  scale:
    min: 0.0
    max: 1.0

  semantics:
    "0.0": >
      No admissible epistemic support.
      Dimension is effectively unsatisfied.
    "0.5": >
      Partial, weak, or incomplete epistemic support.
    "1.0": >
      Fully satisfied epistemic condition
      under declared constitutional constraints.

  rounding:
    precision: 4
    deterministic: true

# -------------------------------------------------------------------
# EPISTEMIC CONFIDENCE DIMENSIONS
# -------------------------------------------------------------------

epistemic_confidence:

  # ===============================================================
  # EVIDENTIARY CONFIDENCE
  # ===============================================================

  evidentiary:
    description: >
      Measures whether claims are supported by
      identifiable, citable evidence.
    inputs:
      - provenance_chain
      - citation_count
    constraints:
      min_citations: 1
    replay_dependencies:
      - provenance_chain_hash
    normalization_rule: >
      f(citation_count, evidence_uniqueness)

  # ===============================================================
  # SOURCE CONSENSUS CONFIDENCE
  # ===============================================================

  source_consensus:
    description: >
      Measures agreement across independent sources.
      Disagreement lowers confidence but does not
      invalidate claims.
    inputs:
      - provenance_chain
      - source_overlap
    constraints:
      independence_required: true
    replay_dependencies:
      - provenance_chain_hash
    normalization_rule: >
      f(independent_source_count, contradiction_ratio)

  # ===============================================================
  # REPLAY DETERMINISM CONFIDENCE
  # ===============================================================

  replay_determinism:
    description: >
      Measures whether execution is deterministically
      reproducible under replay.
    inputs:
      - execution_trace
      - replay_hash
    constraints:
      deterministic_required: true
    replay_dependencies:
      - execution_trace_hash
    normalization_rule: >
      1.0 if replay_hash matches exactly
      else 0.0

  # ===============================================================
  # ATTESTATION STRENGTH
  # ===============================================================

  attestation_strength:
    description: >
      Measures support from required attestations.
      During FIRST_EXECUTABLE_PROOF, this dimension
      may be minimal but must be explicit.
    inputs:
      - attestations
    constraints:
      min_attestations: 0
    replay_dependencies:
      - attestation_hashes
    normalization_rule: >
      f(attestation_count, attestation_weight)

  # ===============================================================
  # TEMPORAL STABILITY
  # ===============================================================

  temporal_stability:
    description: >
      Measures whether claims remain stable over
      constitutional time.
      Cross-epoch evaluation is optional at this phase.
    inputs:
      - epoch_id
      - prior_claim_hashes
    constraints:
      requires_cross_epoch_check: false
    replay_dependencies:
      - epoch_id
    normalization_rule: >
      1.0 if no known prior contradiction
      else value < 1.0
